Values.js is a library to get the tints and shades of a colour.

Download .zip file View on Github 
npm install values.js --save
bower install values.js --save

Basic Usage

const Values = require('values.js')
  , color = new Values('#0099ff');

Examples

Get Tints and Shades

new Values('#2ecc71').all(5).forEach(…);

Get Tints

new Values('#9b59b6').tints(10).forEach(…);

Get Shades

new Values('#3498db').shades(10).forEach(…)

Get a single tint by percentage

new Values('#e74c3c').tint(20);

Get a single shade by percentage

new Values('#f1c40f').shade(10);

Extras

Get Brightness

new Values('#ee0').all(10).forEach(function (value) {
  var e = document.createElement('div')
    , brightness = value.getBrightness();

  e.style.backgroundColor = value.hexString();
  e.textContent = brightness;
  e.style.color = (brightness < 50) ? '#fff' : '#000';

  myContainer.appendChild(e);
});

Utils

Values.Utils.isHex(colorString)

Values.Utils.isHEX('09c')     => true
Values.Utils.isHEX('#09c')    => true
Values.Utils.isHEX('#0099cc') => true
Values.Utils.isHEX('09cc')    => false

Values.Utils.isRGB(colorString)

Values.Utils.isRGB('rgb(0,0,0)')    => true
Values.Utils.isRGB('rgba(0,0,0,0)') => true
Values.Utils.isRGB('0,0,0')         => false

Values.Utils.isHSL(colorString)

Values.Utils.isHSL('hsl(198,58%,1%)')      => true
Values.Utils.isHSL('hsla(360,10%,10%, 1)') => true
Values.Utils.isHSl('hsl(361,10%,10%)')     => false
